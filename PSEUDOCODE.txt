FUNCTION dropDisc(col, disc):
    IF col out of range OR top cell not empty: RETURN -1
    FOR r from ROWS-1 down to 0:
        IF grid[r][col] == ' ':
            grid[r][col] = disc
            RETURN r
    RETURN -1
